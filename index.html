01 Overview

The European Nucleotide Archive (ENA) organises sequencing projects in a hierarchy of umbrella projects (containers) and data projects (leaf nodes holding actual sequencing data). Large initiatives such as ERGA and BGE span dozens — sometimes hundreds — of child projects across multiple umbrella layers.

This toolkit addresses a practical problem: given a hand-drawn diagram of known project accessions, is the diagram complete? Are there registered child or parent projects in ENA that the diagram does not show?

Key insight: the ENA XML API exposes <CHILD_PROJECT> and <PARENT_PROJECT> elements for every project accession. Querying each known accession once is sufficient to surface all immediate neighbours — no recursive crawling needed.

02 Pipeline

🖼️
Input
Hand-drawn diagram of known PRJEB accessions
🔍
Crawler
Fetch ENA XML for each seed accession, extract relationships
📊
JSON
Structured graph data with known + missing neighbours
🗺️
Generator
Render clustered DOT graph via Graphviz
Design decision — no BFS expansion: the crawler fetches only the accessions listed in the input diagram. Neighbours outside this seed set are reported as missing but never themselves queried, preventing an unbounded cascade through ENA's full project graph (ERGA alone links transitively to thousands of projects).

Dashed lines indicate relationships that do not yet exist in ENA — in this case the BGE umbrella project, which had not been created at time of analysis.

04 Crawler — ena_project_crawler.py

How it works

For each seed accession the script calls the ENA Browser XML API:

shell
curl "https://www.ebi.ac.uk/ena/browser/api/xml/PRJEB43510"

The response XML lists immediate neighbours under <RELATED_PROJECTS>:

ENA XML — relationship tags
<RELATED_PROJECTS>
  <RELATED_PROJECT>
    <CHILD_PROJECT accession="PRJEB109737"/>
  </RELATED_PROJECT>
  <RELATED_PROJECT>
    <PARENT_PROJECT accession="PRJNA533106"/>
  </RELATED_PROJECT>
</RELATED_PROJECTS>
XML gotcha: the correct tag names are <CHILD_PROJECT> and <PARENT_PROJECT> — not <IS_CHILD> / <IS_PARENT> as some older documentation suggests.

Full source

ena_project_crawler.py
#!/usr/bin/env python3
"""
ENA Umbrella Project Relationship Crawler
==========================================
Queries the EBI ENA API for project relationships (parent/child umbrella links)
starting from a set of known accessions, then discovers new ones recursively.

Usage:
    python ena_project_crawler.py

Output:
    - Console: summary of all projects and relationships found
    - ena_relationships.json: full graph data
    - ena_graph.dot: Graphviz DOT file for visualization
    - ena_missing_from_diagram.txt: accessions/relationships not in the PNG
"""

import urllib.request
import xml.etree.ElementTree as ET
import json
import time
from collections import defaultdict

# ── Known accessions from the PNG diagram ─────────────────────────────────────
KNOWN_FROM_PNG = {
    "PRJEB43510",   # ERGA
    "PRJEB78703",   # BGE genome skimming
    "PRJEB108398",  # BGE Metabarcoding Project
    "PRJEB106061",  # Insect pollinators T12.2-3-7
    "PRJEB61747",   # ERGA-BGE
    "PRJEB109737",  # ERGA-BGE APPLICATIONS
    "PRJEB108399",  # Bulk Sampling
    "PRJEB108401",  # High Mountains
    "PRJEB108396",  # Pollinator
    "PRJEB108476",  # Water Sampling
    "PRJEB108475",  # Soil Sampling
    "PRJEB105536",  # WP11
    "PRJEB105601",  # WP11
    # Note: one WP11 accession is cut off in the PNG (PRJEB1059??)
}

# Dashed (not-yet-created) BGE umbrella relationships shown in PNG
DASHED_RELATIONSHIPS = {
    ("BGE", "PRJEB78703"),   # BGE → BGE genome skimming
    ("BGE", "PRJEB108398"),  # BGE → BGE Metabarcoding Project
    ("BGE", "PRJEB106061"),  # BGE → Insect pollinators
    ("BGE", "PRJEB61747"),   # BGE → ERGA-BGE  (side link)
}

ENA_API = "https://www.ebi.ac.uk/ena/browser/api/xml/{accession}"


# ── Fetch & parse one project ─────────────────────────────────────────────────

def fetch_project_xml(accession: str) -> ET.Element | None:
    url = ENA_API.format(accession=accession)
    try:
        with urllib.request.urlopen(url, timeout=15) as resp:
            raw = resp.read()
        return ET.fromstring(raw)
    except Exception as exc:
        print(f"  [WARN] Could not fetch {accession}: {exc}")
        return None


def parse_project(accession: str, root: ET.Element) -> dict:
    """Extract name, type, children, and parents from an ENA XML response."""
    info = {
        "accession": accession,
        "name": "",
        "is_umbrella": False,
        "children": [],   # child project accessions
        "parents": [],    # parent project accessions
    }

    project_el = root.find(".//PROJECT")
    if project_el is None:
        return info

    # Name
    name_el = project_el.find("NAME")
    title_el = project_el.find("TITLE")
    info["name"] = (name_el.text if name_el is not None else
                    title_el.text if title_el is not None else accession)

    # Umbrella flag
    umbrella_el = project_el.find(".//UMBRELLA_PROJECT")
    info["is_umbrella"] = umbrella_el is not None

    # Child projects (this project IS an umbrella that lists children)
    for child in project_el.findall(".//CHILDREN/PROJECT"):
        acc = child.get("accession") or child.get("refname")
        if acc:
            info["children"].append(acc.strip())

    # Related projects section (may list parents as well as children)
    for rel in project_el.findall(".//RELATED_PROJECTS/RELATED_PROJECT"):
        # <IS_PARENT><PROJECT accession="..."/></IS_PARENT>
        parent_el = rel.find("IS_PARENT/PROJECT")
        if parent_el is not None:
            acc = parent_el.get("accession")
            if acc:
                info["parents"].append(acc.strip())

        child_el = rel.find("IS_CHILD/PROJECT")
        if child_el is not None:
            acc = child_el.get("accession")
            if acc:
                info["children"].append(acc.strip())

    # De-duplicate
    info["children"] = list(dict.fromkeys(info["children"]))
    info["parents"]  = list(dict.fromkeys(info["parents"]))
    return info


# ── Crawl ─────────────────────────────────────────────────────────────────────

def crawl(seeds: set[str]) -> dict[str, dict]:
    """BFS over ENA project graph starting from seed accessions."""
    visited: dict[str, dict] = {}
    queue = list(seeds)

    while queue:
        acc = queue.pop(0)
        if acc in visited:
            continue

        print(f"  Fetching {acc} …")
        root = fetch_project_xml(acc)
        if root is None:
            visited[acc] = {"accession": acc, "name": "?", "is_umbrella": False,
                            "children": [], "parents": []}
            continue

        info = parse_project(acc, root)
        visited[acc] = info

        # Queue newly discovered accessions
        for neighbour in info["children"] + info["parents"]:
            if neighbour not in visited:
                queue.append(neighbour)

        time.sleep(0.3)   # be polite to EBI servers

    return visited


# ── Graph helpers ─────────────────────────────────────────────────────────────

def build_edge_set(projects: dict[str, dict]) -> set[tuple[str, str]]:
    """Return canonical (parent, child) edge set from crawled data."""
    edges = set()
    for acc, info in projects.items():
        for child in info["children"]:
            edges.add((acc, child))
        for parent in info["parents"]:
            edges.add((parent, acc))
    return edges


def find_missing(projects: dict[str, dict], edges: set[tuple[str, str]]) -> dict:
    known_accs = KNOWN_FROM_PNG
    all_accs   = set(projects.keys())

    new_accs   = all_accs - known_accs - {"BGE"}   # BGE not created yet

    # Edges that connect at least one known node but aren't dashed (not-yet) ones
    dashed_normalized = {(p, c) for p, c in DASHED_RELATIONSHIPS
                         if p != "BGE" and c != "BGE"}
    png_edges = set()  # we reconstruct what PNG implies from known_accs
    # Every edge between two known accessions is "in PNG"
    for p, c in edges:
        if p in known_accs and c in known_accs:
            png_edges.add((p, c))

    missing_edges = edges - png_edges
    return {
        "new_accessions": sorted(new_accs),
        "missing_edges":  sorted(missing_edges),
        "all_edges":      sorted(edges),
    }


# ── Output writers ────────────────────────────────────────────────────────────

def write_dot(projects: dict[str, dict], edges: set[tuple[str, str]],
              filename: str = "ena_graph.dot"):
    lines = ["digraph ENA_Projects {",
             '  rankdir=BT;',
             '  node [fontname="Helvetica", fontsize=10];',
             '  edge [color="#555555"];',
             ""]

    for acc, info in sorted(projects.items()):
        label = f"{info['name']}\\n{acc}"
        if info["is_umbrella"]:
            style = 'shape=box, style=filled, fillcolor="#AED6F1"'
        else:
            style = 'shape=box, style=filled, fillcolor="#FAD7A0"'
        if acc not in KNOWN_FROM_PNG:
            style += ', color=red, penwidth=2'   # highlight new nodes
        lines.append(f'  "{acc}" [{style}, label="{label}"];')

    lines.append("")
    for parent, child in sorted(edges):
        color = "red" if (parent not in KNOWN_FROM_PNG or
                          child  not in KNOWN_FROM_PNG) else "#555555"
        lines.append(f'  "{parent}" -> "{child}" [color="{color}"];')

    lines.append("}")
    with open(filename, "w") as f:
        f.write("\n".join(lines))
    print(f"\n[OK] DOT file written → {filename}")


def write_report(projects: dict[str, dict], missing: dict,
                 filename: str = "ena_missing_from_diagram.txt"):
    lines = [
        "ENA Project Graph – Missing from PNG diagram",
        "=" * 60,
        "",
        "ACCESSIONS NOT IN PNG (newly discovered):",
    ]
    if missing["new_accessions"]:
        for acc in missing["new_accessions"]:
            info = projects.get(acc, {})
            umbrella = "umbrella" if info.get("is_umbrella") else "data"
            lines.append(f"  {acc}  [{umbrella}]  {info.get('name','?')}")
    else:
        lines.append("  (none – diagram appears complete)")

    lines += ["", "EDGES NOT SHOWN IN PNG (excluding dashed/future BGE links):"]
    if missing["missing_edges"]:
        for p, c in missing["missing_edges"]:
            pname = projects.get(p, {}).get("name", "?")
            cname = projects.get(c, {}).get("name", "?")
            lines.append(f"  {p} ({pname})  →  {c} ({cname})")
    else:
        lines.append("  (none)")

    lines += ["", "ALL DISCOVERED EDGES:"]
    for p, c in missing["all_edges"]:
        lines.append(f"  {p} → {c}")

    with open(filename, "w") as f:
        f.write("\n".join(lines))
    print(f"[OK] Report written → {filename}")


# ── Main ──────────────────────────────────────────────────────────────────────

def main():
    print("=" * 60)
    print("ENA Umbrella Project Crawler")
    print("=" * 60)
    print(f"\nStarting BFS from {len(KNOWN_FROM_PNG)} seed accessions …\n")

    projects = crawl(KNOWN_FROM_PNG)
    edges    = build_edge_set(projects)
    missing  = find_missing(projects, edges)

    # Console summary
    print("\n" + "=" * 60)
    print(f"Total projects fetched : {len(projects)}")
    print(f"Total edges found      : {len(edges)}")
    print(f"New accessions (not in PNG): {len(missing['new_accessions'])}")
    for acc in missing["new_accessions"]:
        info = projects.get(acc, {})
        print(f"  *** {acc}  {'[umbrella]' if info.get('is_umbrella') else '[data]'}  "
              f"{info.get('name','?')}")

    print(f"\nEdges missing from PNG : {len(missing['missing_edges'])}")
    for p, c in missing["missing_edges"]:
        print(f"  {p} → {c}")

    # Write outputs
    with open("ena_relationships.json", "w") as f:
        json.dump({"projects": projects, "edges": list(edges)}, f, indent=2)
    print("\n[OK] JSON data written → ena_relationships.json")

    write_dot(projects, edges, "ena_graph.dot")
    write_report(projects, missing, "ena_missing_from_diagram.txt")

    print("\nTo render the DOT file (requires Graphviz):")
    print("  dot -Tpng ena_graph.dot -o ena_graph.png")
    print("  dot -Tsvg ena_graph.dot -o ena_graph.svg")
    print("\nDone.")


if __name__ == "__main__":
    main()

05 Graph Generator — ena_graph_generator.py

Approach

Rather than rendering every individual unknown child project as a node (ERGA-BGE alone has 303 unknown children, which produces an unreadable single-line layout), the generator collapses them into a single summary node per parent:

DOT — collapsed summary node
"ERGABGE_extra" [
  label="+303 genome/assembly\ndata projects\n(not in PNG)",
  fillcolor="#FADBD8", color=red,
  style="filled,rounded,dashed"
];

Umbrella projects with known children are wrapped in Graphviz subgraph cluster_* blocks, drawing a visible bounding box around each family of nodes.

Usage

shell
# PNG output (default)
python ena_graph_generator.py

# SVG — recommended for large graphs (scales infinitely)
python ena_graph_generator.py --svg

# Custom JSON path or higher DPI
python ena_graph_generator.py --json /path/to/ena_relationships.json --dpi 300

ena_graph_generator.py
#!/usr/bin/env python3
"""
ENA Graph Generator
====================
Reads ena_relationships.json and produces ena_graph.dot + ena_graph.png

Requirements:
    - Python 3.8+
    - Graphviz installed (brew install graphviz)

Usage:
    python ena_graph_generator.py
    python ena_graph_generator.py --json path/to/ena_relationships.json
    python ena_graph_generator.py --svg   # produce SVG instead of PNG
"""

import json
import subprocess
import argparse
from collections import defaultdict
from pathlib import Path


def load_data(json_path: str) -> tuple:
    with open(json_path) as f:
        data = json.load(f)
    projects        = data["projects"]
    missing_nb      = data["missing_neighbours"]
    return projects, missing_nb


def summarise_missing(missing_nb: dict) -> tuple:
    """Split missing neighbours into unknown children (per parent) and unknown parents."""
    unknown_children = defaultdict(list)   # parent_acc -> [child_accs]
    unknown_parents  = {}                  # acc -> meta

    for acc, meta in missing_nb.items():
        if meta["role"] == "child":
            for parent in meta["referenced_by"]:
                unknown_children[parent].append(acc)
        else:
            unknown_parents[acc] = meta

    return dict(unknown_children), unknown_parents


def build_dot(projects: dict, unknown_children: dict, unknown_parents: dict) -> str:
    lines = [
        "digraph ENA_Projects {",
        "  rankdir=TB;",
        "  compound=true;",
        '  graph [fontname="Helvetica", nodesep=0.5, ranksep=1.0, pad=0.6];',
        '  node  [fontname="Helvetica", fontsize=10, shape=box, style="filled,rounded", margin="0.2,0.12"];',
        '  edge  [fontname="Helvetica", fontsize=9, arrowsize=0.75];',
        "",
    ]

    # ── Unknown parent node(s) ────────────────────────────────────────────────
    for acc, meta in sorted(unknown_parents.items()):
        lines.append(
            f'  "{acc}" [label="{acc}\\n(parent — not in PNG)", '
            f'fillcolor="#FADBD8", color=red, penwidth=2, style="filled,rounded,dashed"];'
        )
    if unknown_parents:
        lines.append("")

    # ── ERGA top-level cluster ────────────────────────────────────────────────
    erga_extra = len(unknown_children.get("PRJEB43510", []))
    lines += [
        "  subgraph cluster_ERGA {",
        '    label="The European Reference Genome Atlas (ERGA)\\nPRJEB43510";',
        '    fontname="Helvetica-Bold"; fontsize=12;',
        '    style="filled,rounded"; fillcolor="#EBF5FB"; color="#2E86C1"; penwidth=2;',
        '    "PRJEB43510" [label="ERGA\\nPRJEB43510", fillcolor="#AED6F1"];',
    ]
    if erga_extra:
        lines.append(
            f'    "ERGA_extra" [label="+{erga_extra} other children\\n(not in PNG)", '
            f'fillcolor="#FADBD8", color=red, penwidth=1.5, style="filled,rounded,dashed", fontsize=9];'
        )
    lines += ["  }", ""]

    # ── ERGA-BGE cluster ──────────────────────────────────────────────────────
    ergabge_extra = len(unknown_children.get("PRJEB61747", []))
    lines += [
        "  subgraph cluster_ERGABGE {",
        '    label="ERGA-BGE\\nPRJEB61747";',
        '    fontname="Helvetica-Bold"; fontsize=11;',
        '    style="filled,rounded"; fillcolor="#EAF4FB"; color="#5DADE2"; penwidth=1.5;',
        '    "PRJEB61747" [label="ERGA-BGE\\nPRJEB61747", fillcolor="#AED6F1"];',

    if ergabge_extra:
        lines.append(
            f'    "ERGABGE_extra" [label="+{ergabge_extra} genome/assembly\\ndata projects\\n(not in PNG)", '
            f'fillcolor="#FADBD8", color=red, penwidth=1.5, style="filled,rounded,dashed", fontsize=9];'
        )
    lines += ["  }", ""]

    # ── ERGA-BGE APPLICATIONS cluster ─────────────────────────────────────────
    apps_extra = len(unknown_children.get("PRJEB109737", []))
    lines += [
        "  subgraph cluster_APPS {",
        '    label="ERGA-BGE APPLICATIONS\\nPRJEB109737";',
        '    fontname="Helvetica-Bold"; fontsize=11;',
        '    style="filled,rounded"; fillcolor="#FEF9E7"; color="#F39C12"; penwidth=1.5;',

    # ── BGE-metabarcoding cluster (contains Bulk-Sampling sub-cluster) ────────
    lines += [
        "  subgraph cluster_META {",
        '    label="BGE-metabarcoding\\nPRJEB108398";',
        '    fontname="Helvetica-Bold"; fontsize=11;',
        '    style="filled,rounded"; fillcolor="#F4ECF7"; color="#8E44AD"; penwidth=1.5;',
        '    "PRJEB108398" [label="BGE-metabarcoding\\nPRJEB108398", fillcolor="#AED6F1"];',
        '    "PRJEB108475" [label="Ecological Restoration\\nSoil - Chronosequences\\nPRJEB108475", fillcolor="#FAD7A0"];',
        '    "PRJEB108476" [label="Marine Invasive\\nSpecies\\nPRJEB108476", fillcolor="#FAD7A0"];',
        "",

        '      "PRJEB108399" [label="Bulk-Sampling\\nPRJEB108399", fillcolor="#AED6F1"];',

        '      "PRJEB108401" [label="HMS\\nPRJEB108401", fillcolor="#FAD7A0"];',
        "    }",
        "  }",

    ]

    # ── Orphan nodes (no known parent — BGE umbrella not yet created) ─────────
    lines += [
        '  "PRJEB78703"  [label="BGE genome skimming\\nPRJEB78703",  fillcolor="#FAD7A0"];',
        '  "PRJEB106061" [label="BGE - WP12\\nInsect Pollinators\\nPRJEB106061", fillcolor="#FAD7A0"];',
        "",
    ]

    # ── Edges: known relationships ────────────────────────────────────────────
    known_edges = [
        ("PRJNA533106", "PRJEB43510",  "red",     "dashed"),  # unknown parent
        ("PRJEB43510",  "PRJEB61747",  "#333333", "solid"),
        ("PRJEB43510",  "PRJEB109737", "#333333", "solid"),
        ("PRJEB109737", "PRJEB105536", "#333333", "solid"),

    for src, dst, color, style in known_edges:

    # ── Edges: to collapsed unknown-child summary nodes ───────────────────────
    lines.append("")
    lines.append("  // Collapsed unknown-children summary edges")
    summary_edges = [
        ("PRJEB43510",  "ERGA_extra",    erga_extra),

        if count:
            lines.append(f'  "{src}" -> "{dst}" [color=red, style=dashed];')

    lines.append("}")
    return "\n".join(lines)


def render(dot_path: str, out_format: str = "png", dpi: int = 150):
    out_path = dot_path.replace(".dot", f".{out_format}")
    cmd = ["dot", f"-T{out_format}"]
    if out_format == "png":
        cmd += [f"-Gdpi={dpi}"]
    cmd += [dot_path, "-o", out_path]

    result = subprocess.run(cmd, capture_output=True, text=True)
    if result.returncode != 0:
        print(f"[ERROR] Graphviz: {result.stderr}")
    else:
        print(f"[OK] Rendered → {out_path}")
    return out_path


def main():
    parser = argparse.ArgumentParser(description="Generate ENA project graph")
    parser.add_argument("--json", default="ena_relationships.json",
                        help="Path to ena_relationships.json (default: ./ena_relationships.json)")
    parser.add_argument("--svg", action="store_true",
                        help="Output SVG instead of PNG")
    parser.add_argument("--dpi", type=int, default=150,
                        help="DPI for PNG output (default: 150)")
    args = parser.parse_args()

    if not Path(args.json).exists():
        print(f"[ERROR] File not found: {args.json}")
        print("  Run ena_project_crawler_v3.py first to generate it.")
        return

    print(f"Loading {args.json} …")
    projects, missing_nb = load_data(args.json)
    unknown_children, unknown_parents = summarise_missing(missing_nb)

    print(f"  Known projects   : {len(projects)}")
    print(f"  Unknown children : { {k: len(v) for k, v in unknown_children.items()} }")
    print(f"  Unknown parents  : {list(unknown_parents.keys())}")

    dot_str  = build_dot(projects, unknown_children, unknown_parents)
    dot_path = "ena_graph.dot"
    with open(dot_path, "w") as f:
        f.write(dot_str)
    print(f"[OK] DOT written  → {dot_path}")

    fmt = "svg" if args.svg else "png"
    render(dot_path, fmt, args.dpi)


if __name__ == "__main__":
    main()
